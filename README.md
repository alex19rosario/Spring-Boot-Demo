## This is the Oracle Database script to create tables for roles, users, and one business entity.
```sql
DROP TABLE BOOKS;
DROP TABLE AUTHORITIES;
DROP TABLE USERS;
DROP SEQUENCE BOOKS_SEQUENCE;

CREATE TABLE USERS (
    USERNAME NVARCHAR2(128) CONSTRAINT USER_ID_PK PRIMARY KEY,
    PASSWORD NVARCHAR2(128) NOT NULL,
    ENABLED CHAR(1) CONSTRAINT USER_ENABLED_CK CHECK (ENABLED IN ('Y','N') ) NOT NULL
);

CREATE TABLE AUTHORITIES (
    USERNAME NVARCHAR2(128) NOT NULL,
    AUTHORITY NVARCHAR2(128) NOT NULL
);
ALTER TABLE AUTHORITIES ADD CONSTRAINT AUTHORITIES_UNIQUE UNIQUE (USERNAME, AUTHORITY);
ALTER TABLE AUTHORITIES ADD CONSTRAINT AUTHORITIES_FK1 FOREIGN KEY (USERNAME) REFERENCES USERS (USERNAME) ENABLE;
```
